CSV file

parsing: using rules to understand a file as structured data
CSV: comma separated values

import csv
f = open(<path>)
csv_f = csv.reader(f)
for row in csv_f:
    a, b, c = row #unpack
    print("{};{};{}".format(a, b, c))
f.close()
list1 = [["a", "b"], ["c", "d"]]
with open(<path>, "w") as f2write:
    writer = csv.writer(f2write)
    writer.writerow(list1) # write 1 row
    writer.writerows(list1) # write all
f = open(<path>)
reader = csv.DictReader(f) #read csv as a row to a dict
for row in reader:
    print(row[<key>])
list1 = [{"a":1, "b":2, "c":3}, {"a":4, "b":5, "c":6}]
keys = ["a", "b", "c"]
with open(<path>, "w") as f2w:
    writer = csv.DictWriter(f2w, fieldnames=keys)
    writer.writeheader() # first row, by keys
    writer.writerows(list1)
