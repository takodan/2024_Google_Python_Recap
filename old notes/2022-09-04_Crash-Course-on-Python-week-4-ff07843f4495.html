string reference cheat sheet

formatting strings cheat sheet

#integer value
'{:d}'.format(10.5) → '10'
#floating point with that many decimals
'{:.2f}'.format(0.5) → '0.50'
#string with that many characters
'{:.2s}'.format('Python') → 'Py'
#string aligned to the left that many spaces
'{:<6s}'.format('Py') → 'Py    '
#string aligned to the right that many spaces
'{:>6s}'.format('Py') → '    Py'
#string centered in that many spaces
'{:^6s}'.format('Py') → '  Py  '

lists are mutable.

list1 = ["a", "b", "c", "d"]
list1.append("w") #["a", "b", "c", "d", "w"]
list1.insert(0, "x") #["x", "a", "b", "c", "d", "w"]
list1.insert(20, "y") #["x", "a", "b", "c", "d", "w", "y"]
list1.remove("a") # ["x", "b", "c", "d", "w", "y"]
list1.pop(2) # ["x", "b", "d", "w", "y"] return "c"
list1.sort()
list1.reverse()
list1.clear()
list1.copy()
list1.extend(<other_list>)

tuples are immutable similar to lists.

tuple1 = ("a", "b", "c")
a, b, c = tuple1 # a = "a", b = "b", c = "c", call unpacking

the functions that return multiple values are usually tuples.
they are immutable so order matters.

enumerate can be used in sequence

list1 = ["a", "b", "c", "d"]
for index, element in enumerate(list1):
    print(index, element, end=";") # 0 a;1 b;2 c;3 d;

list comprehensions
create new lists based on sequences or ranges

mutiples = [x*7 for x in range(1,6)] #[7, 14, 21, 28, 35]
z = [x for x in range(0, 11) if x % 3 == 0] #[0, 3, 6, 9]
