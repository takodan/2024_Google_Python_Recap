regular expression

regex/regexp: find a string of text that match a pattern

in a command line:

grep -i <string> <path> # find filename match string under a path
# -i mean regardless of case

reserved characters

"." #any character
grep l.rts <path> # alerts, blurts, flirts
"^" #caret. a string start with
grep ^fruit <path> # fruit, fruitcake, fruited
"$" #a string end with
grep cat$ <path> # bobcat, cat, ducat

basic

import re
result = re.search(r"aza", "plaza") #r<string>: raw string
print(result)
# <re.Match object; span=(2,5), match='aza'>
# if doesn't match, return None
# can use reserved characters in string
print(re.search(r"p.ng", "Pangaea", re.IGNORECASE))
# <re.Match object; span=(0,4), match='Pang')>

wildcard: in regex, characters match more than one character. like “.”
characters classes: [<characters>]

print(re.search(r"[a-z]way", "the highway")
# <re.Match object; span=(7,11), match='hway')>
print(re.search(r"[a-zA-Z0-9]way", "the way")
# None
[^a] #not a
[a|b] # a or b
# re.search() only get first one
# re.findall() return all found in a list

repetition qualifiers

# *: repeat any time, include 0
re.search(r"py.*n", "python wah ing" # match='python wah in'
re.search(r"py[a-z]*n", "python wah ing") # match='python'
re.search(r"py[a-z]*n", "pyn" # match='pyn'
"a+b+" # ab or aabb or aaabbb... (a least 1 ab)
"a?b" # a is optional(0 or 1 a)
"a*?b+?c??", "aaaabbbbcccc" # match='aaaab', take as little as possible

escaping characters

re.search(r".com", "welcome") # "lcom"
re.search(r"\.com", "welcome") # None
\n # next line (special string character)
\t # tab (special string character)
\w # word (letters, numbers, and underscores)
\d # digits
\s # whitespace (space, tab, or newline)
\b # word boundaries
"^A.*a$" #a word must start with "A" and end with "a"

advanced

capturing groups

>>> import re
>>> result = re.search(r”^(\w*), (\w*)$”, “Hatsune, Mik
>>> print(result)
<re.Match object; span=(0, 13), match=’Hatsune, Miku’>
>>> print(result.groups())
(‘Hatsune’, ‘Miku’)
>>> print(result[0])
Hatsune, Miku
>>> print(result[1])
Hatsune
>>> print(result[2])
Miku

repetition qualifiers

>>> import re
>>> print(re.findall(r"[a-zA-z]{5}", "a scary ghost appeared"))
['scary', 'ghost', 'appea']
>>> print(re.findall(r"\b[a-zA-z]{5}\b", "a scary ghost appeared"))
['scary', 'ghost']
"\w{5,10}" # 5 to 10 words
"\w{5, }" # more then 5
"\w{ ,10}" # up to 10
